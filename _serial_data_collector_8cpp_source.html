311  Dimensions mpi_pos(0, 0, 0);

313  {
314  mpi_pos.set(*mpiPosition);
315  }
316 
317  DCGroup group;
318  group.open(handles.get(mpi_pos), group_path);
319 
320  if (dataName)
321  {
322  // read attribute from the dataset or group
323  hid_t obj_id = -1;

336  } else
337  {
338  throw DCException(getExceptionString("readAttribute",
339  "dataset not found", obj_name.c_str()));
340  }
341  } else

343  // read attribute from the iteration group
344  DCAttribute::readAttribute(attrName, group.getHandle(), data);

360  void SerialDataCollector::writeAttribute(int32_t id,
361  const CollectionType& type,
362  const char *dataName,
363  const char *attrName,
364  uint32_t ndims,
365  const Dimensions dims,
366  const void* data)
367  throw (DCException)
368  {
369  if (attrName == NULL || data == NULL)
370  throw DCException(getExceptionString("writeAttribute", "a parameter was null"));
371 
372  // dataName may be NULL, attribute is attached to iteration group in that case
373  if (dataName && strlen(dataName) == 0)
374  throw DCException(getExceptionString("writeAttribute", "empty dataset name"));
375 
376  if (strlen(attrName) == 0)
377  throw DCException(getExceptionString("writeAttribute", "empty attribute name"));
378 
379  if (fileStatus == FST_CLOSED || fileStatus == FST_READING || fileStatus == FST_MERGING)
380  throw DCException(getExceptionString("writeAttribute", "this access is not permitted"));
381 
382  if (ndims < 1u || ndims > DSP_DIM_MAX)
383  throw DCException(getExceptionString("writeAttribute", "maximum dimension `ndims` is invalid"));
384 
385  /* group_path: absolute path to the last inode
386  * obj_name: last inode, can be a group or a dataset
387  */
388  std::string group_path, obj_name;
389  std::string dataNameInternal = "";
390  if (dataName)
391  dataNameInternal.assign(dataName);
392  DCDataSet::getFullDataPath(dataNameInternal, SDC_GROUP_DATA, id, group_path, obj_name);
393 
394  DCGroup group;
395  if (dataName)
396  {
397  /* if the specified inode (obj_name) does not exist
398  * (as dataset or group), create all missing groups along group_path
399  * and even create an empty group for obj_name itself
400  *
401  * group_path + "/" + obj_name is the absolute path of dataName
402  */
403  std::string pathAndName(group_path + "/" + obj_name);
404  if(!DCGroup::exists(handles.get(id), pathAndName))
405  group.create(handles.get(id), pathAndName);
406 
407  // attach attribute to the dataset or group
408  group.open(handles.get(0), group_path);
409 
410  hid_t obj_id = H5Oopen(group.getHandle(), obj_name.c_str(), H5P_DEFAULT);
411  if (obj_id < 0)
412  {
413  throw DCException(getExceptionString("writeAttribute",
414  "object not found", obj_name.c_str()));
415  }
416 
417  try
418  {
419  DCAttribute::writeAttribute(attrName, type.getDataType(), obj_id, ndims, dims, data);
420  } catch (const DCException&)
421  {
422  H5Oclose(obj_id);
423  throw;
424  }
425  H5Oclose(obj_id);
426  } else
427  {
428  // attach attribute to the iteration group
429  group.openCreate(handles.get(0), group_path);
430  DCAttribute::writeAttribute(attrName, type.getDataType(), group.getHandle(), ndims, dims, data);
431  }
432  }
